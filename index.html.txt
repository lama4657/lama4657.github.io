<script>
    // دالة لوضع مثال توضيحي في الحقول
    function setExample() {
        document.getElementById('expression').value = 'x^2 + 3*x';
        document.getElementById('lowerLimit').value = '0';
        document.getElementById('upperLimit').value = '5';
    }

    // دالة حساب التكامل
    function calculateIntegral() {
        // التأكد من تحميل المكتبة
        if (typeof math === 'undefined') {
            alert("خطأ: يرجى التأكد من الاتصال بالإنترنت لتحميل المكتبة الرياضية.");
            return;
        }

        const expr = document.getElementById('expression').value;
        const a = parseFloat(document.getElementById('lowerLimit').value);
        const b = parseFloat(document.getElementById('upperLimit').value);
        const resultArea = document.getElementById('result-area');
        const output = document.getElementById('output');

        // التحقق من أن الحقول ليست فارغة
        if (!expr || isNaN(a) || isNaN(b)) {
            alert("الرجاء تعبئة جميع الحقول بشكل صحيح");
            return;
        }

        try {
            // إعداد المتغيرات لحساب قاعدة سمبسون (Simpson's Rule)
            const n = 1000; // عدد الشرائح (دقة الحساب)
            const h = (b - a) / n;
            let sum = 0;
            
            // استخدام مكتبة math.js
            const f = (x) => math.evaluate(expr, { x: x });

            sum += f(a) + f(b);

            for (let i = 1; i < n; i++) {
                const x = a + i * h;
                sum += (i % 2 === 0 ? 2 : 4) * f(x);
            }

            const integral = (h / 3) * sum;

            // عرض النتيجة النهائية
            output.innerText = integral.toFixed(4); // تقريب لـ 4 خانات
            resultArea.style.display = 'block'; // إظهار مربع النتيجة

        } catch (error) {
            alert("خطأ في صيغة الدالة! تأكد من استخدام صيغة صحيحة (مثال: 3*x بدلاً من 3x)");
            console.error(error);
        }
    }
</script>
